  geom_col()+
  labs(title="Comparison of drug A",
       x="Patient",
       y="Heartrate")

heartrate %>% 
  pivot_longer(-patient,
                names_to="drug",
                values_to="heartrate") %>% 
  ggplot(aes(x=drug,y=heartrate, fill=drug))+
  geom_boxplot(alpha=0.7)+
  labs(title="Comparison of drug A",
       x="Patient",
       y="Heartrate")
